│   │   ├── device_controller.py  # 设备指令封装、单位转换、限幅
│   │   ├── workflow.py           # 配方执行流程、工位协调
│   │   ├── state_machine.py      # 全局状态机（空闲/供料/脱泡/拉伸/报警）

│   │   └── device_status.py # 设备状态快照（温度、位置、错误码等）
│   └── core/                # 基础设施

│       └── logger.py        # 分级日志+操作审计
